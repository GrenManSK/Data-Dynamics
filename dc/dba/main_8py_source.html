461def search_engine(query, data) -> None | list[str]:
462 query_words = query.lower().split()
463
464 results = []
465 for item in data:
466 for text in item:
467 words = text.lower().replace('"', "").replace("'", "").split()
468 match = True
469 for query_word in query_words:
470 word_matched = False
471 for word in words:
472 distance = Levenshtein.distance(query_word, word)
473 if distance <= 1:
474 word_matched = True
475 break
476 if not word_matched:
477 match = False
478 break
479 if match:
480 results.append(text)
481 break
482
483 if results:
484 if query in results:
485 return [query]
486 return results
487 else:
488 return None
